 26:     /**
 27:      * Callback method for the Google Calendar API authorization process.
 28:      * 
 29:      * Once the user grants consent with his Google Calendar data usage, the Google
 30:      * OAuth service will redirect him back in this page. Here we are going to store
 31:      * the refresh token, because this is what will be used to generate access tokens
 32:      * in the future.
 33:      * 
 34:      * <strong>IMPORTANT!</strong> Because it is necessary to authorize the application
 35:      * using the web server flow (see official documentation of OAuth), every 
 36:      * Easy!Appointments installation should use its own calendar api key. So in every
 37:      * api console account, the "http://path-to-e!a/google/oauth_callback" should be 
 38:      * included in an allowed redirect url.
 39:      */
 40:     public function oauth_callback() {
 41:         if (isset($_GET['code'])) {
 42:             $this->load->library('Google_Sync');
 43:             $token = $this->google_sync->authenticate($_GET['code']);
 44:             
 45:             // Store the token into the database for future reference.
 46:             if (!isset($_SESSION)) {
 47:                 @session_start();
 48:             }
 49:             
 50:             if (isset($_SESSION['oauth_provider_id'])) {
 51:                 $this->load->model('providers_model');
 52:                 
 53:                 $this->providers_model->set_setting('google_sync', TRUE, 
 54:                         $_SESSION['oauth_provider_id']);
 55:                 $this->providers_model->set_setting('google_token', $token, 
 56:                         $_SESSION['oauth_provider_id']);
 57:                 
 58:             } else {
 59:                 echo '<h1>Sync provider id not specified!</h1>';
 60:             }
 61:             
 62:         } else {
 63:             echo '<h1>Authorization Failed!</h1>';
 64:         }

 67:     /**
 68:      * Complete synchronization of appointments between Google Calendar and Easy!Appointments.
 69:      * 
 70:      * This method will completely sync the appointments of a provider with his Google Calendar
 71:      * account. The sync period needs to be relatively small, because a lot of API calls might 
 72:      * be necessary and this will lead to consuming the Google limit for the Calendar API usage.
 73:      * 
 74:      * @param numeric $provider_id Provider record to be synced.
 75:      * 
 76:      * @task This method must be executed only by the system and noone else outside. 
 77:      * It is a big security issue.
 78:      */
 79:     public function sync($provider_id = NULL) {
 80:         try {
 81:             if ($provider_id === NULL) {
 82:                 throw new Exception('Provider id not specified.');
 83:             }
 84:             
 85:             $this->load->model('appointments_model');
 86:             $this->load->model('providers_model');
 87:             $this->load->model('services_model');
 88:             $this->load->model('customers_model');
 89:             $this->load->model('settings_model');
 90:             
 91:             $provider = $this->providers_model->get_row($provider_id);
 92:             
 93:             // Check whether the selected provider has google sync enabled.
 94:             $google_sync = $this->providers_model->get_setting('google_sync', $provider['id']);
 95:             if (!$google_sync) {
 96:                 throw new Exception('The selected provider has not the google synchronization '
 97:                         . 'setting enabled.');
 98:             }
 99:             
100:             $google_token = json_decode($this->providers_model->get_setting('google_token', $provider['id']));
101:             $this->load->library('google_sync');
102:             $this->google_sync->refresh_token($google_token->refresh_token);
103:             
104:             // Fetch provider's appointments that belong to the sync time period. 
105:             $sync_past_days = $this->providers_model->get_setting('sync_past_days', $provider['id']);
106:             $sync_future_days = $this->providers_model->get_setting('sync_future_days', $provider['id']);
107:             $start = strtotime('-' . $sync_past_days . ' days', strtotime(date('Y-m-d')));
108:             $end = strtotime('+' . $sync_future_days . ' days', strtotime(date('Y-m-d')));
109:             
110:             $where_clause = array(
111:                 'start_datetime >=' => date('Y-m-d H:i:s', $start),
112:                 'end_datetime <=' => date('Y-m-d H:i:s', $end),
113:                 'id_users_provider' => $provider['id'],
114:                 'is_unavailable' => FALSE
115:             );
116:             
117:             $appointments = $this->appointments_model->get_batch($where_clause);
118:             
119:             $company_settings = array(
120:                 'company_name' => $this->settings_model->get_setting('company_name'),
121:                 'company_link' => $this->settings_model->get_setting('company_link'),
122:                 'company_email' => $this->settings_model->get_setting('company_email')
123:             );
124:             
125:             // Sync each appointment with Google Calendar by following the project's sync
126:             // protocol (see documentation).
127:             foreach($appointments as $appointment) {
128:                 $service = $this->services_model->get_row($appointment['id_services']);
129:                 $customer = $this->customers_model->get_row($appointment['id_users_customer']);
130: 
131:                 // If current appointment not synced yet, add to gcal.
132:                 if ($appointment['id_google_calendar'] == NULL) {
133:                     $google_event = $this->google_sync->add_appointment($appointment, $provider,
134:                             $service, $customer, $company_settings);
135:                     $appointment['id_google_calendar'] = $google_event->id;
136:                     $this->appointments_model->add($appointment); // Save gcal id
137:                 } else {
138:                     // Appointment is synced with google calendar.
139:                     try {
140:                         $google_event = $this->google_sync->get_event($appointment['id_google_calendar']);
141:                     
142:                         // If gcal event is different from e!a appointment then update e!a record.
143:                         $is_different = FALSE;
144:                         $appt_start = strtotime($appointment['start_datetime']);
145:                         $appt_end = strtotime($appointment['end_datetime']);
146:                         $event_start = strtotime($google_event->getStart()->getDateTime());
147:                         $event_end = strtotime($google_event->getEnd()->getDateTime());
148: 
149:                         if ($appt_start != $event_start || $appt_end != $event_end) {
150:                             $is_different = TRUE;
151:                         }
152: 
153:                         if ($is_different) {
154:                             $appointment['start_datetime'] = date('Y-m-d H:i:s', $event_start);
155:                             $appointment['end_datetime'] = date('Y-m-d H:i:s', $event_end);
156:                             $this->appointments_model->add($appointment);
157:                         }
158:                     } catch(Exception $exc) {
159:                         // Appointment not found on gcal, delete from e!a.
160:                         $this->appointments_model->delete($appointment['id']);
161:                         $appointment['id_google_calendar'] = NULL; 
162:                     }
163:                 }
164:             }
165:             
166:             // :: ADD GCAL EVENTS THAT ARE NOT PRESENT ON E!A
167:             $events = $this->google_sync->get_sync_events($start, $end);
168:             
169:             foreach($events->getItems() as $event) {
170:                 $results = $this->appointments_model->get_batch(array('id_google_calendar' => $event->getId()));
171:                 if (count($results) == 0) {
172:                     // Record doesn't exist in E!A, so add the event now.
173:                     $appointment = array(
174:                         'start_datetime' => date('Y-m-d H:i:s', strtotime($event->start->getDateTime())),
175:                         'end_datetime' => date('Y-m-d H:i:s', strtotime($event->end->getDateTime())),
176:                         'is_unavailable' => TRUE,
177:                         'notes' => $event->getSummary() . ' ' . $event->getDescription(),
178:                         'id_users_provider' => $provider_id,
179:                         'id_google_calendar' => $event->getId(),
180:                         'id_users_customer' => NULL,
181:                         'id_services' => NULL,
182:                     );
183:                     
184:                     $this->appointments_model->add($appointment);
185:                 }
186:             }

194:         }
